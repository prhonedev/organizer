Array Methods

Métodos de objetos array.

// array.concat(item...)
var a = ['a', 'b', 'c'];
var b = ['x', 'y', 'z'];
var c = a.concat(b, true);
// c is ['a', 'b', 'c', 'x', 'y', 'z', true]

// array.join(separator)
var a = ['a', 'b', 'c'];
var b = a.join('');  // c is 'abc'

// array.reverse()
var a = ['a', 'b', 'c'];
var b = a.reverse();
// both a and b are ['c', 'b', 'a']


// array.push(item...)
var a = ['a', 'b', 'c'];
var b = ['x', 'y', 'z'];
var c = a.push(b, true);
// a is ['a', 'b', 'c', ['x', 'y', 'z'], true] && c is 5

// array.unshift(item...)
var a = ['a', 'b', 'c'];
var r = a.unshift('?', '@');
// a is ['?', '@' 'a', 'b', 'c'] && r is 5


// array.pop()
var a = ['a', 'b', 'c'];
var b = a.pop();  // a is ['a', 'b'] && b is 'c'

// array.shift()
var a = ['a', 'b', 'c'];
var b = a.shift();  // a is ['b', 'c'] && b is 'a'


// array.slice(start, endBefore)
var a = ['a', 'b', 'c'];
var b = a.slice(0, 1);  // b is ['a']
var c = a.slice(1);  // c is ['b', 'c']
var d = a.slice(1, 2);  // d is ['b']

// array.splice(start, deleteCount, items...)
var a = ['a', 'b', 'c'];
var r = a.splice(1, 1, 'ache', 'bug');
// a is ['a', 'ache', 'bug', 'c'] && r is ['b']

Regular Expressions

Algunas expresiones regulares comúnes.

// /regexp/g reemplazar todas las ocurrencias
// /regexp/i case-insensitive: se ignoran mayúsculas y minúsculas
var a = 'Romel Francisco Pérez Estrada';
var b = a.replace(/r/gi, '@');
// b is "@omel F@ancisco Pé@ez Est@ada"

Conversión

Conversión números de base 10 a otra base numérica y viceversa.

// Convertir a base 10
var converTo10 = function (n, b) {
    return parseInt(n, b);
};

// Convertir de base10 a otra base
var convertFrom10 = function (n, b) {
    return n.toString(b);
};
